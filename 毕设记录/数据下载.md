```js
// === 缅甸 2025 年 2 月 & 4 月夜光数据（修复波段 & ID 错误）===

// 1. 缅甸边界（用 'Burma' 确保不空）
var roi = ee.FeatureCollection("USDOS/LSIB_SIMPLE/2017")
          .filter(ee.Filter.eq('country_na', 'Burma'))
          .geometry();

// 2. 2025 年 2 月：月度数据集（VCMCFG，已到 2025-03，选择正确波段）
var monthlyCollection = ee.ImageCollection('NOAA/VIIRS/DNB/MONTHLY_V1/VCMCFG')
                         .select('avg_rad')  // 修复：用 'avg_rad' 而非 'DNB_at_sensor_radiance_500nits'
                         .filterBounds(roi);
var feb2025 = monthlyCollection.filterDate('2025-02-01', '2025-03-01')
                               .first()
                               .clip(roi);

// 3. 2025 年 4 月：每日数据集（VNP46A2，合成月度；数据已覆盖 2025 年）
var dailyCollection = ee.ImageCollection('NASA/VIIRS/002/VNP46A2')  // 修复：正确每日 ID
                        .select('Gap_Filled_DNB_BRDF_Corrected_NTL')  // 夜光波段（0-1 范围）
                        .filterDate('2025-04-01', '2025-05-01')
                        .filterBounds(roi);
var apr2025 = dailyCollection.median()  // 中值合成，去云/噪声
                             .clip(roi);

// 4. 导出到 Google Drive（两个 GeoTIFF，~15-25 MB/张）
Export.image.toDrive({
  image: feb2025,
  description: 'Myanmar_Nightlight_2025_02',
  folder: 'GEE_Nightlight_2025',
  region: roi, scale: 500, crs: 'EPSG:4326', maxPixels: 1e13
});

Export.image.toDrive({
  image: apr2025,
  description: 'Myanmar_Nightlight_2025_04_DailyComposite',
  folder: 'GEE_Nightlight_2025',
  region: roi, scale: 500, crs: 'EPSG:4326', maxPixels: 1e13
});

// 5. 可视化（地图显示两层对比，无错误）
Map.centerObject(roi, 7);
Map.addLayer(feb2025, {min: 0, max: 60, palette: ['black', 'purple', 'cyan', 'yellow', 'white']}, '2025年2月（月度官方）');
Map.addLayer(apr2025, {min: 0, max: 1, palette: ['black', 'purple', 'cyan', 'yellow', 'white']}, '2025年4月（每日合成）');  // max:1 匹配 VNP46A2 范围

// 6. Console 确认（打印图像详情和图像数）
print('2025年2月图像:', feb2025);
print('2025年4月合成图像:', apr2025);
print('2月月度图像数（应为1）:', monthlyCollection.filterDate('2025-02-01', '2025-03-01').size());
print('4月每日原始图像数（应>20）:', dailyCollection.size());
```
